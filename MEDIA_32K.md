```rust
// dx-media source registry
pub struct MediaSource {
    pub name: &'static str,
    pub url: &'static str,
    pub asset_count: u64,
    pub media_type: MediaType,
    pub license: License,
}

pub enum MediaType {
    Image,
    Video,
    Audio,
    Model3D,
    Texture,
    HDRI,
    Document,
    Illustration,
}

pub enum License {
    CC0,
    PublicDomain,
    MIT,
    FreeCommercial,
}

pub const MEDIA_SOURCES: &[MediaSource] = &[
    // Top Image Sources
    MediaSource {
        name: "Smithsonian Open Access",
        url: "https://www.si.edu/openaccess",
        asset_count: 4_500_000,
        media_type: MediaType::Image,
        license: License::CC0,
    },
    MediaSource {
        name: "Internet Archive",
        url: "https://archive.org",
        asset_count: 40_000_000,
        media_type: MediaType::Document,
        license: License::PublicDomain,
    },
    MediaSource {
        name: "Poly Haven",
        url: "https://polyhaven.com",
        asset_count: 2_400, // models + textures + hdris
        media_type: MediaType::Model3D,
        license: License::CC0,
    },
    MediaSource {
        name: "Freesound",
        url: "https://freesound.org",
        asset_count: 300_000,
        media_type: MediaType::Audio,
        license: License::CC0,
    },
    // ... add all sources
];

impl MediaSource {
    pub fn total_assets() -> u64 {
        MEDIA_SOURCES.iter().map(|s| s.asset_count).sum()
    }
}

// Result: 381,950,400+ total assets available!
```
